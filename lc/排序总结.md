###### 快速排序

注意点：在递归的时候要判断是否越界

```c++
int partition(int l, int r, vector<int>& nums) {

        int index = rand() % (r-l+1) + l;
        swap(nums[l], nums[index]);
        int x = nums[l];
        while (l < r) {
            while (l < r && x <= nums[r]) r--;
            swap(nums[l], nums[r]);
            //nums[l] = nums[r];
            while (l < r && x >= nums[l]) l++;
            swap(nums[r], nums[l]);
        }
        nums[l] = x;
        return l;
    }
    void quick_sort(int l, int r, std::vector<int>& nums) {
        if (l > r) return;
        int mid = partition(l, r, nums);
        quick_sort(l, mid - 1, nums);
        quick_sort(mid + 1, r, nums);
    }

    vector<int> sortArray(vector<int>& nums) {
        int len = nums.size();
        quick_sort(0, len - 1, nums);
        return nums;
    }
```

###### 归并

```c++
//这里是merge两个有序数组，然后将临时排好序的数据拷贝过去 
void merge(std::vector<int>& nums, int l, int mid, int r) {
        std::vector<int> tmp(r-l+1, 0);
        int i = l, j = mid + 1, k = 0;
        while (i <= mid && j <= r) {
            if (nums[i] < nums[j]) {
                tmp[k++] = nums[i++];
            } else {
                tmp[k++] = nums[j++];
            }
        }
        while (i <= mid) {
            tmp[k++] = nums[i++];
        }
        while (j <= r) {
            tmp[k++] = nums[j++];
        }
        //这里是小于k或者小于r-l+1,不能等于；
        for (int m = 0; m < k; m++) {
            nums[l+m] = tmp[m];
        }

    }
//递归的逐层向下拆分数组，直到一个数组只有一个元素，当有0个元素的时候，
//直接返回
    void merge_sort(std::vector<int>& nums, int l, int r) {
        if (l >= r) return;
        int mid = l + (r-l)/2;
        merge_sort(nums, l, mid);
        merge_sort(nums, mid + 1, r);
        merge(nums, l, mid, r);
    }
    vector<int> sortArray(vector<int>& nums) {
        int len = nums.size();
        //quick_sort(0, len - 1, nums);
        merge_sort(nums, 0, nums.size() - 1);
        return nums;
    }
```



###### 堆排序

由于堆排序是完全二叉树，直接通过数组就可以保存树形结构；数组的下标可以作为树节点的索引，比如：当从0 开始的时候，左孩子是2*i+1；右孩子是2*i + 2;如果给定一个下标i，其父亲节点的的索引是（i-1）/2;

把长度作为限制条件进行递归

```c++
void heapfiy(std::vector<int>& nums, int len, int index)
{
    if (index >= len)   return;
    int max = index;
    int c1 = 2 * index + 1;
    int c2 = 2 * index + 2;
    if (c1 < len && nums[c1] > nums[max]) {
        max = c1;
    }
    if (c2 < len && nums[c2] > nums[max]) {
        max = c2;
    }
    if (index != max) {
        swap(nums[index], nums[max]);
        heapfiy(nums, len, max);
    }
}
//在建堆的时候从最后一个节点的父节点开始调整，从底到顶，一次调整好；
void build_heap(std::vector<int>& nums)
{
    int last_index = nums.size() - 1;
    int parent = (last_index - 1) / 2;
    for (int i = parent; i >= 0; i--) {
        heapfiy(nums, nums.size(), i);
    }
}
void heap_sort(std::vector<int>& nums) {
    build_heap(nums);
    int len = nums.size();
    for (int i = len - 1; i >= 0; i--) {
        //交换了之后从跟节点开始调整，一直到交换前，此时的i是索引，也代表当前调整的节点的数量
        swap(nums[i], nums[0]);
        heapfiy(nums, i, 0);
        
    }
}
```

总结：在leetcod上进行提交的时候，发现归并用了空间和时间最多，由于中间有临时数组的拷贝，所以导致性能很差，在工作中要注意
